package controller;

import dao.VoyageDAO;
import entity.Voyage;
import java.time.LocalDateTime;
import java.util.List;

public class VoyageController {
    private final VoyageDAO voyageDAO = new VoyageDAO();

    // CREATE
    public boolean creerVoyage(String destination, LocalDateTime dateHeureDepart,
                             LocalDateTime dateHeureArrivee, double prix,
                             String aeroportDepart, String aeroportArrivee,
                             int nbrPlaces) {
        Voyage voyage = new Voyage(
            0, // ID auto-généré
            destination,
            dateHeureDepart,
            dateHeureArrivee,
            prix,
            aeroportDepart,
            aeroportArrivee,
            nbrPlaces
        );
        return voyageDAO.create(voyage);
    }

    // READ
    public List<Voyage> listerTousVoyages() {
        return voyageDAO.readAll();
    }

    public Voyage getVoyageParId(int id) {
        return voyageDAO.getById(id);
    }

    // UPDATE
    public boolean modifierVoyage(Voyage voyage) {
        return voyageDAO.update(voyage);
    }

    // DELETE
    public boolean supprimerVoyage(int id) {
        return voyageDAO.delete(id);
    }

    // Méthode supplémentaire pour la gestion des places
    public boolean mettreAJourPlaces(int idVoyage, int nouvellesPlaces) {
        Voyage voyage = voyageDAO.getById(idVoyage);
        if(voyage != null) {
            voyage.setNbrPlaces(nouvellesPlaces);
            return voyageDAO.update(voyage);
        }
        return false;
    }
}
