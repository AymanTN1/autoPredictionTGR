package dao;

import entity.Voyage;
import java.sql.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

public class VoyageDAO {
    
    // CREATE
    public boolean create(Voyage voyage) {
        String sql = "INSERT INTO Voyage (destination, date_depart, periode_jours, prix, "
                   + "aeroport_depart, aeroport_arrivee, date_heure_depart, "
                   + "date_heure_arrivee, nbr_places) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)";
        
        try (Connection conn = db.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            setVoyageParameters(pstmt, voyage);
            return pstmt.executeUpdate() > 0;
            
        } catch (SQLException e) {
            handleException("Erreur création voyage", e);
            return false;
        }
    }

    // READ
    public List<Voyage> readAll() {
        List<Voyage> voyages = new ArrayList<>();
        String sql = "SELECT * FROM Voyage";
        
        try (Connection conn = db.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {
            
            while (rs.next()) {
                voyages.add(mapResultSetToVoyage(rs));
            }
        } catch (SQLException e) {
            handleException("Erreur lecture voyages", e);
        }
        return voyages;
    }

    public Voyage getById(int id) {
        String sql = "SELECT * FROM Voyage WHERE id = ?";
        try (Connection conn = db.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setInt(1, id);
            ResultSet rs = pstmt.executeQuery();
            
            if (rs.next()) {
                return mapResultSetToVoyage(rs);
            }
        } catch (SQLException e) {
            handleException("Erreur recherche voyage", e);
        }
        return null;
    }

    // UPDATE
    public boolean update(Voyage voyage) {
        String sql = "UPDATE Voyage SET destination = ?, date_depart = ?, periode_jours = ?, "
                   + "prix = ?, aeroport_depart = ?, aeroport_arrivee = ?, "
                   + "date_heure_depart = ?, date_heure_arrivee = ?, nbr_places = ? "
                   + "WHERE id = ?";
        
        try (Connection conn = db.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            setVoyageParameters(pstmt, voyage);
            pstmt.setInt(10, voyage.getId());
            
            return pstmt.executeUpdate() > 0;
            
        } catch (SQLException e) {
            handleException("Erreur mise à jour voyage", e);
            return false;
        }
    }

    // DELETE
    public boolean delete(int id) {
        String sql = "DELETE FROM Voyage WHERE id = ?";
        try (Connection conn = db.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setInt(1, id);
            return pstmt.executeUpdate() > 0;
            
        } catch (SQLException e) {
            handleException("Erreur suppression voyage", e);
            return false;
        }
    }

    // Méthodes utilitaires
    private Voyage mapResultSetToVoyage(ResultSet rs) throws SQLException {
        return new Voyage(
            rs.getInt("id"),
            rs.getString("destination"),
            rs.getTimestamp("date_heure_depart").toLocalDateTime(),
            rs.getTimestamp("date_heure_arrivee").toLocalDateTime(),
            rs.getDouble("prix"),
            rs.getString("aeroport_depart"),
            rs.getString("aeroport_arrivee"),
            rs.getInt("nbr_places")
        );
    }

    private void setVoyageParameters(PreparedStatement pstmt, Voyage voyage) throws SQLException {
        pstmt.setString(1, voyage.getDestination());
        pstmt.setDate(2, Date.valueOf(voyage.getDateHeureDepart().toLocalDate()));
        pstmt.setInt(3, voyage.getPeriodeJours());
        pstmt.setDouble(4, voyage.getPrix());
        pstmt.setString(5, voyage.getAeroportDepart());
        pstmt.setString(6, voyage.getAeroportArrivee());
        pstmt.setTimestamp(7, Timestamp.valueOf(voyage.getDateHeureDepart()));
        pstmt.setTimestamp(8, Timestamp.valueOf(voyage.getDateHeureArrivee()));
        pstmt.setInt(9, voyage.getNbrPlaces());
    }

    private void handleException(String message, SQLException e) {
        System.err.println(message + " : " + e.getMessage());
        e.printStackTrace();
    }
}
