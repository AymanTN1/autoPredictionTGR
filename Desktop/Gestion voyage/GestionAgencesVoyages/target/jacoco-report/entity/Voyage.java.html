package entity;

import java.time.LocalDateTime;

public class Voyage {
    private int id;
    private String destination;
    private LocalDateTime dateHeureDepart;
    private LocalDateTime dateHeureArrivee;
    private int periodeJours;
    private double prix;
    private String aeroportDepart;
    private String aeroportArrivee;
    private int nbrPlaces;

    public Voyage(int id, String destination, LocalDateTime dateHeureDepart, 
                LocalDateTime dateHeureArrivee, double prix, String aeroportDepart, 
                String aeroportArrivee, int nbrPlaces) {
        this.id = id;
        this.destination = destination;
        this.dateHeureDepart = dateHeureDepart;
        this.dateHeureArrivee = dateHeureArrivee;
        this.periodeJours = (int) java.time.Duration.between(dateHeureDepart, dateHeureArrivee).toDays();
        this.prix = prix;
        this.aeroportDepart = aeroportDepart;
        this.aeroportArrivee = aeroportArrivee;
        this.nbrPlaces = nbrPlaces;
    }

    // Getters
    public int getId() { return id; }
    public String getDestination() { return destination; }
    public LocalDateTime getDateHeureDepart() { return dateHeureDepart; }
    public LocalDateTime getDateHeureArrivee() { return dateHeureArrivee; }
    public int getPeriodeJours() { return periodeJours; }
    public double getPrix() { return prix; }
    public String getAeroportDepart() { return aeroportDepart; }
    public String getAeroportArrivee() { return aeroportArrivee; }
    public int getNbrPlaces() { return nbrPlaces; }

    // Setters
    public void setDestination(String destination) { this.destination = destination; }
    public void setDateHeureDepart(LocalDateTime dateHeureDepart) { 
        this.dateHeureDepart = dateHeureDepart;
        this.periodeJours = (int) java.time.Duration.between(dateHeureDepart, dateHeureArrivee).toDays();
    }
    public void setDateHeureArrivee(LocalDateTime dateHeureArrivee) { 
        this.dateHeureArrivee = dateHeureArrivee;
        this.periodeJours = (int) java.time.Duration.between(dateHeureDepart, dateHeureArrivee).toDays();
    }
    public void setPrix(double prix) { this.prix = prix; }
    public void setAeroportDepart(String aeroportDepart) { this.aeroportDepart = aeroportDepart; }
    public void setAeroportArrivee(String aeroportArrivee) { this.aeroportArrivee = aeroportArrivee; }
    public void setNbrPlaces(int nbrPlaces) { this.nbrPlaces = nbrPlaces; }
}
